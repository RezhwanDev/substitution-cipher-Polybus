20 public void enc(){
21     System.out.println("Enter Text ");
22     Scanner sob = new Scanner(System.in);
23     String s = sob.nextLine();
24     s.toUpperCase();
25     for (int i = 0; i < s.length(); i++) {
26         for (int j = 0; j < 5; j++) {
27             for (int k = 0; k < 5; k++) {
28                 if (s.charAt(i) == alpha[j][k]) {
